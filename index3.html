```javascript int curveCount = 0; // 현재까지 그려진 곡선의 개수 int maxCurves = 500; // 총 곡선 개수 void setup() { size(800, 800); // 캔버스 크기 설정 background(255); // 흰색 배경 frameRate(60); // 초당 60프레임 설정 (1초에 최대 60개의 곡선) } void draw() { if (curveCount < maxCurves) { drawRandomCurve(); curveCount++; } else { noLoop(); // 모든 곡선을 그린 후 중지 } } void drawRandomCurve() { // 랜덤 색상 생성 stroke(random(100, 255), random(50, 200), random(150, 255), random(50, 150)); strokeWeight(random(1, 5)); // 랜덤 두께 설정 noFill(); // 곡선의 시작점과 끝점 설정 float x1 = random(width); float y1 = random(height); float x2 = random(width); float y2 = random(height); // 제어점 설정 (베지어 곡선의 형태를 만듦) float cx1 = random(width); float cy1 = random(height); float cx2 = random(width); float cy2 = random(height); // 베지어 곡선 그리기 bezier(x1, y1, cx1, cy1, cx2, cy2, x2, y2); }
